Boas Praticas

querSelectorAll = // Traz um array de todos os objetos var teste = document.querySelectorAll(".objeto"); for (var i = 0; teste.length; i++){ console.log(teste[i]); }

buscar pelo objeto

var valorx = document.querySelector(.titulo); // busca pela classe var valorx = document.querySelector(#titulo); // busca pelo ID

Pegando valores de tabela e somando var y = document.querySelector("#conteudo TR"); var z = y.querySelector(".conteudo TD"); var val = z.textContent; // valor da TD

adicionar atributos aos objetos

var valorx = document.querySelector(.titulo); // busca pela classe valorx.classList.add("nome de uma classe qualquer ou inclusive de um css")

Adicionar um escutador, evento

var valorx = document.querySelector(.titulo); // busca pela classe valorx.addEventListener("click", mostraMensagem); function mostraMensagem(){ alert("Mensagem"); }

OU funcao anonima

valorx.addEventListener("click", function() { alert("Mensagem"); });
